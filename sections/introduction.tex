\section{Introduction} 

\begin{frame}[label=stockpred]{Why Predict Stock Prices?}
	\begin{itemize} 	
		\item Stock prediction offers \emph{high potential returns}
        \item Financial markets are \alert{noisy and nonlinear}
        \item Forecasting aids decision-making
        \item Hybrid models \alert{(LSTM + GRU) improve accuracy and efficiency}
        \item \emph{Technical indicators enhance deep learning performance}
	\end{itemize}
	\note{
        Stock market prediction has always attracted attention due to its potential for significant 
        financial gain. Historically, investors used fundamental and technical analysis to guide 
        decisions, but these traditional methods often struggle with the high volatility and complexity 
        of financial time series.
        
        With advances in deep learning, models like LSTM have gained traction. LSTMs are a type of 
        recurrent neural network specifically designed to retain long-term dependencies in sequential 
        data, making them well-suited for modeling stock prices.

        These models are further enhanced when combined with technical indicators—such as MACD, RSI, 
        Bollinger Bands, and SMA—which provide insight into historical trends and market momentum.
        
        Furthermore, combining LSTM with GRU—a simpler, more efficient recurrent unit—has shown promise in 
        improving both training speed and predictive accuracy. This project explores the effectiveness of 
        LSTM and hybrid LSTM–GRU architectures in forecasting stock prices, specifically for Workday Inc.
    }
\end{frame}

\begin{frame}[label=problemstatement]{Problem Statement }
    \begin{block}{Research Focus}
        \begin{itemize}
            \item Forecast \alert{next-day closing price} of \href{https://www.nasdaq.com/market-activity/stocks/wday}{Workday Inc. (WDAY)} using deep learning
            \item Address limitations of traditional methods in volatile, nonlinear financial markets
        \end{itemize}
    \end{block}

    \begin{alertblock}{Objective}
        Develop and compare LSTM and LSTM-BiGRU models to minimize prediction errors, identify short-term trends, and support informed decision-making.
    \end{alertblock}

    \note{
        This project focuses on forecasting the next-day stock closing price of Workday Inc. (WDAY) using advanced deep learning techniques.
        Given the increasing volatility and complexity of financial markets, traditional approaches often fail to capture the nonlinear dynamics involved.

        The core objective is to develop and compare two deep learning architectures—LSTM and BiGRU—in predicting WDAY's next-day closing price.
        The goal is not only to minimize forecasting errors but also to uncover actionable short-term trends.

        Additionally, this research explores how such deep learning forecasts could be integrated into real-time applications like algorithmic trading systems or financial monitoring tools.
    }
\end{frame}
